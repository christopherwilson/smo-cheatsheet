\subsubsection*{Discrete Random Variables}
\wde{Bernoulli A.3} Let $X\sim\mathrm{Bernoulli}(p)$, then $P(X = 1) = p$, $P(X=0) = 1-p$, $\E(X) = p$, $\Var(X) = p(1-p)$.
\wde{Binomial A.3} Let $X\sim\mathrm{Binomial}(n, p) = \sum^{n}_{i=1} \mathrm{Bernoulli}(p)$, then $P(X = 1) = \binom{n}{i}p^i(1-p)^{n-i}$, $\E(X) = np$, $\Var(X) = np(1-p)$.
\wde{Poisson A.3} Let $X \sim \mathrm{Poisson}(\lambda)$, which is the $n \to \infty$ limit of a $\mathrm{Binomial}(n,p)$ with $\lambda = np$, $P(X = n) = e^{-\lambda}\lambda^n/n!$, $\E(x) = \lambda$, $\Var(X) = \lambda$.
\wde{Geometric 4.3} $\mathrm{geometric}(p)$: Number of independent $\mathrm{Bernoulli}(p)$ trials till first success, $P(X = n) = (1-p)^{n-1}p$, $\E(X) = 1/p$, $\Var(X) = (1-p)/p^2$.
\wde{Negative Binomial 4.3} Let $X \sim\ $negative binomial$(n,p) = \sum^{r}_{i = 1} \mathrm{geometric}(p)$, the number of independent $\mathrm{Bernoulli}(p)$ trials till the $r$-th success, $P(X = n) = \binom{n-1}{r-1}(1-p)^{n-r} p^r$, $\E(X) = r/p$, $\Var(X) = r(1-p)/p^2$.
\wde{Hypergeometric 4.3} Let $X \sim \mathrm{hypergeometrix}(N,m,n)$: the number of red balls drawn ehen we draw $n$ balls out of $m$ red balls and $N-m$ blue balls, $P(X=i) = \binom{m}{i}\binom{N-m}{N-i}/\binom{N}{n}$, $\E(X) = nm/N$, $\Var(X) = np(1-p)[1-(n-1)/(N-1)]$, with $p = m/N$.
